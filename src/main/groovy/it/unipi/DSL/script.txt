RIPEAtlas {
    obtain traceroute_list
    from anchors
    to anchors
    limit 3 on measurements
    limit 10 on targets
    limit 10 on sources
    limit 50 on results
    filter hop that ">=" of 3    
    source_country "IT"
    target_country "DE" 
    verbose on
    cache off
    timeframe "2022-01-01 00:00:00",start
    timeframe "2022-05-30 00:00:00",stop
    execute()
}
