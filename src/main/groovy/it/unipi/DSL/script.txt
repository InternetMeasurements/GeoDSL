RIPEAtlas {
    obtain traceroute_list
    from anchors
    to anchors
    limit 1 on measurements 
    limit 5 on targets 
    limit 5 on sources
    limit 20 on results
    filter star that "<=" of 0
    coordinate on
    asn on 
    source_country "IT"
    target_country "DE" 
    verbose on
    cache off
    timeframe "2023-01-01 00:00:00",start
    timeframe "2023-08-27 00:00:00",stop

    execute()
}
